-- | Abstract ledger implementation as described in TZIP-FA1.

module Indigo.Contracts.AbstractLedger
  ( Parameter (..)
  , Storage (..)
  , abstractLedgerContract
  ) where

import Indigo
import Lorentz.Contracts.Spec.AbstractLedgerInterface

data Storage = Storage
  { ledger      :: BigMap Address Natural
  , totalSupply :: Natural
  }
  deriving stock Generic
  deriving anyclass IsoValue

----------------------------------------------------------------------------
-- Entrypoints
----------------------------------------------------------------------------

transfer
  :: HasStorage Storage
  => Var TransferParams
  -> IndigoProcedure s
transfer tp = do
  doc $ DDescription "Transfers tokens between two given accounts."
  debitFrom tp
  creditTo tp

debitFrom ::
  ( tp :~> TransferParams , HasStorage Storage)
  => tp
  -> IndigoProcedure s
debitFrom tp = do
  -- Sender check.
  let
    fromAddr :: Expr Address
    fromAddr = tp !. #from

    failWrongSenderArg :: Expr ("sender" :! Address, "from" :! Address)
    failWrongSenderArg = pair
      (Sender !~ #sender)
      (fromAddr !~ #from)

    failWrongSender :: IndigoProcedure s
    failWrongSender = failCustom #nonAcceptableSource failWrongSenderArg

  if fromAddr /=. Sender then failWrongSender else return ()

  -- Balance check and update.
  let
    ledger_ :: Expr (BigMap Address Natural)
    ledger_ = storage !. #ledger

    val :: Expr Natural
    val = tp !. #value

  let
    failNotEnoughBalance :: ex :~> Natural => ex -> IndigoProcedure s
    failNotEnoughBalance curBalance = failCustom #notEnoughBalance $ pair
      (val !~ #required)
      (curBalance !~ #present)

    setBalance :: ex :~> Maybe Natural => ex -> IndigoProcedure s
    setBalance bal =
      setField_ storage #ledger $ ledger_ &?~ (fromAddr, bal)

    whenExists
      :: Var Natural
      -> IndigoProcedure (Natural ': s)
    whenExists oldBalance = scope $ do
      newBalance <- subGt0 oldBalance val (failNotEnoughBalance oldBalance)
      setBalance newBalance

  ifJust (ledger_ &? fromAddr)
    whenExists
    (failNotEnoughBalance (0 :: Natural))

creditTo ::
  ( tp :~> TransferParams , HasStorage Storage)
  => tp
  -> IndigoProcedure s
creditTo tp = scope $ do
  let
    toAddr :: Expr Address
    toAddr = tp !. #to

    val :: Expr Natural
    val = tp !. #value

    ledger_ :: Expr (BigMap Address Natural)
    ledger_ = storage !. #ledger

  let
    setBalance bal = setField_ storage #ledger $ ledger_ &~ (toAddr, bal)

  setBalance =<< ifJust (ledger_ &? toAddr) (return . (val +.)) (return val)

getTotalSupply
  :: (HasSideEffects, HasStorage Storage)
  => Var (View () Natural)
  -> IndigoProcedure s
getTotalSupply v = do
  doc $ DDescription "Returns total number of tokens."
  view_ (\_ -> return (storage !. #totalSupply)) v

getBalance ::
  ( HasSideEffects , HasStorage Storage)
  => Var (View GetBalanceParams Natural)
  -> IndigoProcedure s
getBalance v = scope $ do
  doc $ DDescription "Returns the balance of the address in the ledger."
  flip view_ v $ \owner -> do
    varExpr <$>
      ifJust (storage !. #ledger &? UnName #owner owner)
             return
             (return (0 :: Natural))

-------------------------------------------------------------------------------
-- Contract
----------------------------------------------------------------------------

abstractLedgerContract :: ContractCode Parameter Storage
abstractLedgerContract = compileIndigoContract abstractLedgerIndigo

storage :: HasStorage Storage => Var Storage
storage = storageVar

abstractLedgerIndigo
  :: (HasStorage Storage, HasSideEffects)
  => Var Parameter
  -> IndigoProcedure '[ Parameter, Storage, Ops]
abstractLedgerIndigo param =  contractName "Abstract Ledger" $ do
  doc $ $mkDGitRevision morleyRepoSettings
  doc $ DDescription contractDoc
  entryCaseSimple param
    ( #cTransfer //-> transfer
    , #cGetTotalSupply //-> getTotalSupply
    , #cGetBalance //-> getBalance
    )
  where
    contractDoc :: Markdown
    contractDoc =
      "This documentation describes a minimal abstract ledger contract " <>
      "that is described in TZIP-FA1."
