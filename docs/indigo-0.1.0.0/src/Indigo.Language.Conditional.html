{-# OPTIONS_GHC -Wno-redundant-constraints #-}

-- | Conditional statements of Indigo language.

module Indigo.Language.Conditional
  ( if_
  , ifJust
  , when
  , whenJust
  , whenNothing
  , IfConstraint
  ) where

import qualified Data.Kind as Kind
import qualified GHC.TypeLits as Lit

import Indigo.Expr
import Indigo.Language.Scope
import Indigo.Lorentz
import Indigo.Prelude
import Indigo.State (GenCode(..), IndigoM(..), TypeValue, Var, pushRefMd, return)
import qualified Lorentz.Instr as L
import qualified Lorentz.Macro as L

type family CompareBranchesResults (a :: Kind.Type) (b :: Kind.Type) :: Constraint where
  CompareBranchesResults x x = ()
  CompareBranchesResults x y = Lit.TypeError
      ('Lit.Text " Result types of if branches diverged: "
       'Lit.:<>: 'Lit.ShowType x 'Lit.:<>: ('Lit.Text " against ") 'Lit.:<>: 'Lit.ShowType y
      )

type IfConstraint a b inp =
  ( ScopeCodeGen a
  , ScopeCodeGen b
  , CompareBranchesResults (ScopeRetExprs a) (ScopeRetExprs b)
  -- This constraint is satisfied by one above, but GHC needs a proof
  , ScopeOutStack a inp ~ ScopeOutStack b inp
  )

-- | If statement. It's supposed that all created
-- variables inside branches will be released after execution
-- leaves a scope where they were created.
if_ :: forall inp xs ys a b exc .
      ( IfConstraint a b inp
      , exc :~> Bool
      )
  => exc
  -> IndigoM inp xs a
  -> IndigoM inp ys b
  -> IndigoM inp (ScopeOutStack a inp) (ScopeRetVars a)
if_ e t f = IndigoM $ \md ->
  let cde = gcCode $ runIndigoM (compileToExpr e) md in
  let gc1 = runIndigoM t md in
  let gc2 = runIndigoM f md in
  finalizeStatement @a md (cde # L.if_ (compileScope gc1) (compileScope gc2))

-- | Run the instruction when condition is met, do nothing otherwise.
when :: (exc :~> Bool) => exc -> IndigoM inp out () -> IndigoM inp inp ()
when cond expr = if_ cond expr (return ())

-- | If which works like case for Maybe.
ifJust :: forall inp xs ys x a b exa .
  ( IfConstraint a b inp, TypeValue x
  , exa :~> Maybe x
  )
  => exa
  -> (Var x -> IndigoM (x & inp) xs a)
  -> IndigoM inp ys b
  -> IndigoM inp (ScopeOutStack a inp) (ScopeRetVars a)
ifJust e t f = IndigoM $ \md ->
  let cde = gcCode $ runIndigoM (compileToExpr e) md in
  let (v, mdJust) = pushRefMd md in
  let gc1 = runIndigoM (t v) mdJust in
  let gc2 = runIndigoM f md in
  finalizeStatement @a md $
    cde #
    L.ifSome
      ( compileScope gc1 #
       -- after this we have stack (e1 & e2 .. & ek & x & inp)
       liftClear' @(ClassifyBranchReturn a) @a @(x & inp) @inp L.drop
       -- this can be lifted together with glClear code, but let's leave it like this for now
      )
      (compileScope gc2)

-- | Run the instruction when the given expression returns `Just`
-- value or do nothing otherwise.
whenJust :: forall inp xs x exa .
     ( TypeValue x
     , exa :~> Maybe x
     )
  => exa
  -> (Var x -> IndigoM (x & inp) xs ())
  -> IndigoM inp inp ()
whenJust c f = ifJust c f (return ())

-- | Run the instruction when the given expression returns `Nothing`
-- or do nothing otherwise.
whenNothing :: forall inp x ys exa .
     ( TypeValue x
     , exa :~> Maybe x
     )
  => exa
  -> IndigoM inp ys ()
  -> IndigoM inp inp ()
whenNothing c f = ifJust c (\_ -> return ()) f
