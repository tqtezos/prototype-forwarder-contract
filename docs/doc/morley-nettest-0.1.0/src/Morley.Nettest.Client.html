-- | Implementation that works with real Tezos network, it
-- talks to a Tezos node and uses `tezos-client`.

module Morley.Nettest.Client
  ( NettestClientConfig (..)
  , NettestClientError (..)
  , runNettestClient
  , nettestImplClient
  ) where

import qualified Data.Text as T
import Fmt (Buildable(..), fixedF, fmt, pretty)
import System.Environment (setEnv)
import System.Exit (ExitCode(..))
import System.Process (readProcessWithExitCode)

import Lorentz.Print
import Michelson.Typed.Arith (ArithErrorType (..))
import Michelson.Typed.EntryPoints
import Tezos.Address

import Morley.Nettest.Abstract

-- | Configuration of nettest implementation for real network.
data NettestClientConfig = NettestClientConfig
  { nccScenarioName :: Maybe Text
  , nccNodeAddress :: Maybe Text
  , nccNodePort :: Maybe Word16
  , nccTezosClient :: FilePath
  , nccNodeUseHttps :: Bool
  , nccVerbose :: Bool
  } deriving stock Show

data NettestClientError
  = UnexpectedClientFailure Text
  | UnknownAddress
      Text  -- ^ Name of address which is eventually used
      Text  -- ^ Original name passed by user
  | UnexpectedClientSuccess
  deriving stock (Show, Eq)

instance Exception NettestClientError where
  displayException = pretty

instance Buildable NettestClientError where
  build = \case
    UnexpectedClientFailure errOutput ->
      "tezos-client unexpectedly failed\n" <> build errOutput
    UnknownAddress name origName ->
      "Could not find an address with name " <> build name <>
      "(original name: " <> build origName <> ")"
    UnexpectedClientSuccess ->
      "tezos-client should have failed, but succedeed instead"

-- | Run 'NettestScenario' using implementation for real network.
-- Throws an error if an unexpected failure occurs or if an expected one doesn't
runNettestClient :: NettestClientConfig -> NettestScenario -> IO ()
runNettestClient conf scenario = do
  -- Disable warning saying that it is not mainnet
  setEnv "TEZOS_CLIENT_UNSAFE_DISABLE_DISCLAIMER" "YES"
  scenario (nettestImplClient conf)

-- | Implementation that works with real network and uses `tezos-node`
-- RPC and `tezos-client`.
nettestImplClient :: NettestClientConfig -> NettestImpl IO
nettestImplClient NettestClientConfig {..} =
  NettestImpl
    { niOriginate = \OriginateData {..} -> do
        void $ callTezosClient
          [ "originate"
          , "contract", toString (prefixName odName)
          , "transferring", showAsXTZ odBalance
          , "from", toString (formatAddrOrAlias odFrom)
          , "running", toString (printLorentzContract True odContract)
          , "--init", toString (printLorentzValue True odStorage)
          , "--force"
          , "--burn-cap", "100" -- LARGE VALUE, DO NOT USE BLINDLY IN MAINNET
          ]
        niResolveAddr (AddrAlias odName)

    , niTransfer = \TransferData {..} -> do
        void $ callTezosClient $ appendEntrypointArg tdEntrypoint
          [ "transfer", showAsXTZ tdAmount
          , "from", toString (formatAddrOrAlias tdFrom)
          , "to", toString (formatAddrOrAlias tdTo)
          , "--arg", toString (printLorentzValue True tdParameter)
          , "--burn-cap", "100" -- LARGE VALUE, DO NOT USE BLINDLY IN MAINNET
          ]
    , niComment = putTextLn
    , niExpectFailure = withExpectedFailure
    , ..
    }
  where
    extraArgs :: [String]
    extraArgs =
      bool id ("-S":) nccNodeUseHttps $
      mconcat $ catMaybes
      [ nccNodeAddress <&> \nodeAddress -> ["-A", toString nodeAddress]
      , nccNodePort <&> \nodePort -> ["-P", show nodePort]
      ]

    callTezosClient' :: Maybe (Text -> Bool) -> [String] -> IO Text
    callTezosClient' expectedErrors args = toText <$> do
      let
        allArgs = extraArgs ++ args

      logDebug $ "Calling with args: " <> show allArgs
      readProcessWithExitCode (toString nccTezosClient) allArgs ""  >>= \case
        (ExitSuccess, output, errOutput) ->
          output <$ putErrLn errOutput
        (ExitFailure _, output, toText -> errOutput) ->
          case expectedErrors of
            Just classifier | classifier errOutput ->
              output <$ putErrLn errOutput
            _ -> case lookupNettestError errOutput of
              Just nettestFailure -> throwM nettestFailure
              Nothing -> throwM $ UnexpectedClientFailure errOutput

    -- Call and expect success
    callTezosClient :: [String] -> IO Text
    callTezosClient = callTezosClient' Nothing

    -- Call and permit some failures
    callTezosClientLenient :: (Text -> Bool) -> [String] -> IO Text
    callTezosClientLenient classifier = callTezosClient' (Just classifier)

    -- Names visible to users of nettest will be prefixed with
    -- "<nettest>" and optional scenario name before being passed to
    -- 'tezos-client'.
    prefixName :: Text -> Text
    prefixName name
      | name == "nettest" = name
      | otherwise =
        T.intercalate "." $ "nettest" : catMaybes [nccScenarioName, Just name]

    formatAddrOrAlias :: AddrOrAlias -> Text
    formatAddrOrAlias = \case
      AddrResolved addr -> formatAddress addr
      AddrAlias name -> prefixName name

    niResolveAddr = \case
      AddrResolved addr -> pure addr
      AddrAlias origName -> do
        let name = prefixName origName
        logDebug $ "Resolving " <> name
        output <- callTezosClient ["list", "known", "contracts"]
        case safeHead (mapMaybe (T.stripPrefix (name <> ": ")) (lines output)) of
          Nothing -> throwM $ UnknownAddress name origName
          Just addrText -> either
            (error . mappend ("tezos-client produced address that we consider invalid: " <> addrText) . pretty)
            pure $
            parseAddress addrText

    niGenKey origName = do
      let
        name = prefixName origName
        isAlreadyExistsError :: Text -> Bool
        -- We can do a bit better here using more complex parsing if necessary.
        isAlreadyExistsError = T.isInfixOf "already exists."

      _ <-
        callTezosClientLenient isAlreadyExistsError
        ["gen", "keys", toString name]
      niResolveAddr (AddrAlias origName)

    niGenFreshKey origName = do
      let
        name = prefixName origName
        isNoAliasError :: Text -> Bool
        -- We can do a bit better here using more complex parsing if necessary.
        isNoAliasError = T.isInfixOf "no public key hash alias named"

      _ <-
        callTezosClientLenient isNoAliasError
        ["forget", "address", toString name, "--force"]

      callTezosClient ["gen", "keys", toString name]
      niResolveAddr (AddrAlias origName)

    logDebug :: Text -> IO ()
    logDebug = when nccVerbose . putErrLn . ("<nettest> " <>)

----------------------------------------------------------------------------
-- Helpers
----------------------------------------------------------------------------

-- | Write something to stderr.
putErrLn :: Print a => a -> IO ()
putErrLn = hPutStrLn stderr

-- This function treats input value as microXTZ and turns it into a
-- string with XTZ.
showAsXTZ :: Word64 -> String
showAsXTZ n = fmt $ fixedF @Double 6 (realToFrac n / 1e6)

appendEntrypointArg :: EpName -> [String] -> [String]
appendEntrypointArg = \case
  DefEpName -> id
  EpNameUnsafe txt -> (++ ["--entrypoint", toString txt])

----------------------------------------------------------------------------
-- Validation
----------------------------------------------------------------------------

-- | Checks that a failure occurs (if one is expected) and that it matches the
-- expected one.
withExpectedFailure :: IO a -> NettestFailure -> IO ()
withExpectedFailure sc ntFailure =
  catch (sc >> throwM UnexpectedClientSuccess) (checkFailure ntFailure)
  where
    checkFailure :: NettestFailure -> NettestFailure -> IO ()
    checkFailure expected raised
      | expected == raised = return ()
      | otherwise = throwM raised

-- | Finds a 'NettestFailure' (if any exist) from the error output of a client call
lookupNettestError :: Text -> Maybe NettestFailure
lookupNettestError errMsg =
  let msgMap =
        [ ("script reached FAILWITH instruction", NettestFailedWith)
        , ("Underflowing subtraction", NettestMutezArithError SubUnderflow)
        , ("Overflowing multiplication", NettestMutezArithError MulOverflow)
        , ("Overflowing addition", NettestMutezArithError AddOverflow)
        , ("unexpected arithmetic overflow", NettestShiftArithError)
        -- NOTE ^ this seems to also cover the shift's underflow error
        , ("Cannot pay storage storage fee", NettestGasExhaustion)
        ]
  in snd <$> find (\(msg, _) -> T.isInfixOf msg errMsg) msgMap
