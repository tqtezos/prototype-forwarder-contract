-- | Common command line options for nettest.

module Morley.Nettest.CLI
  (
    -- * Client config
    getClientConfig
  , clientConfigParser

    -- * Smaller parser
  , scenarioNameOption
  , nodeAddressOption
  , nodePortOption
  , tezosClientOption
  , useTlsFlag
  , verboseFlag
  ) where

import Options.Applicative (Parser, auto, help, long, option, optional, short, strOption)
import qualified Options.Applicative as Opt

import Morley.Nettest.Client

getClientConfig :: Maybe Text -> IO NettestClientConfig
getClientConfig scenarioName =
  Opt.execParser $ Opt.info parser $ mconcat
  [ Opt.briefDesc
  , Opt.progDesc $ toString $
    "Network testing" <> maybe "" (" for " <>) scenarioName
  ]
  where
    parser = clientConfigParser scenarioName <**> Opt.helper

-- | Standard command line options parser for 'NettestClientConfig'.
clientConfigParser :: Maybe Text -> Parser NettestClientConfig
clientConfigParser scenarioName = do
  nccScenarioName <- (<|> scenarioName) <$> scenarioNameOption
  nccNodeAddress <- nodeAddressOption
  nccNodePort <- nodePortOption
  nccTezosClient <- tezosClientOption
  nccNodeUseHttps <- useTlsFlag
  nccVerbose <- verboseFlag
  pure NettestClientConfig {..}

scenarioNameOption :: Parser (Maybe Text)
scenarioNameOption = optional $ strOption $ mconcat
  [ long "scenario-name"
  , help "Override scenario name"
  ]

nodeAddressOption :: Parser (Maybe Text)
nodeAddressOption = optional $ strOption $ mconcat
  [ short 'A'
  , long "node-address"
  , help "Network address of the node to connect to"
  ]

nodePortOption :: Parser (Maybe Word16)
nodePortOption = optional $ option auto $ mconcat
  [ short 'P'
  , long "node-port"
  , help "Network port of the node to connect to"
  ]

tezosClientOption :: Parser FilePath
tezosClientOption = strOption $ mconcat
  [ long "tezos-client"
  , help "Path to the `tezos-client' executable"
  , Opt.value "tezos-client"
  , Opt.showDefault
  ]

useTlsFlag :: Parser Bool
useTlsFlag = Opt.switch $
  short 'S' <>
  long "tls" <>
  help "Use TLS to connect to the node"

verboseFlag :: Parser Bool
verboseFlag = Opt.switch $
  short 'v' <>
  long "verbose" <>
  help "If enabled, the output will be verbose and contain debug log"
