whirlpoolInit :: Context Whirlpool
whirlpoolInit = Context 0 0 B.empty initHash

{-# NOINLINE whirlpoolUpdate #-}
whirlpoolUpdate :: Context Whirlpool -> ByteString -> Context Whirlpool
whirlpoolUpdate ctx@(Context n k w hv) s
  | B.null s               = ctx
  | sizeRead  < sizeToRead = Context (n + fromIntegral sizeRead) (k + sizeRead) (w <> s1) hv
  | sizeRead >= sizeToRead = whirlpoolUpdate (Context (n + fromIntegral sizeToRead) 0 mempty (whirlpoolBlockUpdate hv (w <> s1)) ) s'
  where
    !sizeToRead  = whirlpoolBlockSize - k
    !s1          = B.take sizeToRead s
    !s'          = B.drop sizeToRead s
    !sizeRead    = B.length s1

{-# NOINLINE whirlpoolFinal #-}
whirlpoolFinal :: Context Whirlpool -> Whirlpool
whirlpoolFinal (Context n _ w hv) = foldl' whirlpoolBlockUpdate hv (lastChunk n w)

{-# NOINLINE whirlpoolHash #-}
whirlpoolHash :: LBS.ByteString -> Whirlpool
whirlpoolHash = whirlpoolFinal . LBS.foldlChunks whirlpoolUpdate whirlpoolInit

instance HashAlgorithm Whirlpool where
  hashBlockSize = const whirlpoolBlockSize
  hashDigestSize = const whirlpoolDigestSize
  hashInit = whirlpoolInit
  hashUpdate = whirlpoolUpdate
  hashFinal = whirlpoolFinal
