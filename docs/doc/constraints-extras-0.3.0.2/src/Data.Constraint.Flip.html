{-# LANGUAGE CPP #-}

{-# LANGUAGE FlexibleInstances #-}

{-# LANGUAGE PolyKinds #-}

{-# LANGUAGE UndecidableSuperClasses #-}

module Data.Constraint.Flip
  ( FlipC
  ) where

import Data.Constraint

-- | Flip for constraints.
class c h g => FlipC (c :: k -> k' -> Constraint) (g :: k') (h :: k)
instance c h g => FlipC c g h
