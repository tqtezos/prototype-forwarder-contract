-- | Various errors that can happen in `morley-client`
module Morley.Client.Error
  ( PreApplyError (..)
  ) where

import Fmt (Buildable(..), pretty, (+|), (|+))
import qualified Text.Show (show)

import Morley.Client.Types
import Tezos.Address

-- | Errors that can happen during operations preapply
data PreApplyError
  = PreApplyEmptyResult
  | PreApplyFailed [RunError]
  | PreApplyUnexpectedSizeResult Int Int
  | PreApplyOriginatedMoreContracts [Address]

instance Buildable PreApplyError where
  build = \case
    PreApplyEmptyResult -> "Preapply didn't return any results"
    PreApplyFailed errs -> "Preapply failed due to the following errors:\n" +|
      mconcat (map ((<> "\n\n") . build) errs) |+ ""
    PreApplyUnexpectedSizeResult expected got ->
      "Preapply returned " +| got |+ " results, but " +| expected |+
      " results was expected"
    PreApplyOriginatedMoreContracts addresses ->
      "Operation expected to originate one contract, but will more:\n" +|
      mconcat (map ((<> "\n") . build) addresses) |+ ""

instance Show PreApplyError where
  show = pretty

instance Exception PreApplyError where
  displayException = pretty
