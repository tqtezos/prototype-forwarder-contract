-- | Create a new variable with passed expression as an initial value.
newVar :: (IsExpr ex x, TypeValue x) => ex -> IndigoM inp (x & inp) (Var x)
newVar e = compileToExpr e >> makeTopVar

-- | Set to variable new value.
setVar :: (IsExpr ex x, TypeValue x) => Var x -> ex -> IndigoM inp inp ()
setVar v e = do
  MetaData s _ <- iget
  unaryOpFlat e $ varActionSet v s

-- | Call binary operator with constant argument to update variable in-place.
updateVar
  :: (IsExpr ey y, TypeValue x, TypeValue y)
  => [y, x] :-> '[x]
  -> Var x
  -> ey
  -> IndigoM inp inp ()
updateVar action var e = do
  MetaData s _ <- iget
  unaryOpFlat e $ varActionUpdate var s action

(+=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Add n m
     , ArithResHs M.Add n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(+=) = updateVar L.add

(-=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Sub n m
     , ArithResHs M.Sub n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(-=) = updateVar L.sub

(*=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Mul n m
     , ArithResHs M.Mul n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(*=) = updateVar L.mul

(||=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Or n m
     , ArithResHs M.Or n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(||=) = updateVar L.or

(&&=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.And n m
     , ArithResHs M.And n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(&&=) = updateVar L.and

(^=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Xor n m
     , ArithResHs M.Xor n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(^=) = updateVar L.xor

(<<=)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Lsl n m
     , ArithResHs M.Lsl n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(<<=) = updateVar L.lsl

(>>=.)
  :: ( IsExpr ex1 n
     , ArithOpHs M.Lsr n m
     , ArithResHs M.Lsr n m ~ m
     , TypeValue m
     , TypeValue n
     ) => Var m -> ex1 -> IndigoM inp inp ()
(>>=.) = updateVar L.lsr

-- Can be implemented in this way:
-- setField_ v l newVal = setVar v (SetField (V v) l newVal)
-- but it takes 2x more instructions.
setField_ ::
  forall dt name inp ex . (InstrSetFieldC dt name, TypeValue dt, IsExpr ex (GetFieldType dt name), TypeValue (GetFieldType dt name))
  => Var dt -> Label name -> ex -> IndigoM inp inp ()
setField_ v l e = do
  MetaData s _ <- iget
  unaryOpFlat e $ varActionUpdate v s (L.setField l)

-- Loop
----------------------------------------------------------------------------

-- | While statement. The same rule about releasing.
while :: forall inp xs . Expr Bool -> IndigoM inp xs () -> IndigoM inp inp ()
while e body = IndigoM $ \md ->
  let expCd = gcCode $ runIndigoM (compileExpr e) md in
  let bodyCode = cleanGenCode $ runIndigoM body md in
  GenCode () md (expCd # L.loop (bodyCode # expCd)) L.nop

-- | For statements to iterate over container.
forEach
  :: forall a e inp xs. (IterOpHs a, TypeValue (IterOpElHs a), e :~> a)
  => e -> (Var (IterOpElHs a) -> IndigoM ((IterOpElHs a) & inp) xs ())
  -> IndigoM inp inp ()
forEach container body = IndigoM $ \md ->
  let cde = gcCode $ runIndigoM (compileToExpr container) md in
  let (var, newMd) = pushRefMd md in
  let bodyCode = cleanGenCode $ runIndigoM (body var) newMd in
  GenCode () md (cde # L.iter (bodyCode # L.drop)) L.nop

----------------------------------------------------------------------------
-- Documentation
----------------------------------------------------------------------------

-- | Put a document item.
doc :: DocItem di => di -> IndigoM s s ()
doc di = IndigoM \md -> GenCode () md (L.doc di) L.nop

-- | Group documentation built in the given piece of code
-- into block dedicated to one thing, e.g. to one entrypoint.
docGroup :: DocGrouping -> IndigoM i o () -> IndigoM i o ()
docGroup gr ii = IndigoM $ \md ->
  let GenCode _ mdii cd clr = runIndigoM ii md in
  GenCode () mdii (L.docGroup gr cd) clr

selfCalling
  :: forall inp p mname.
     ( NiceParameterFull p
     , TypeValue (GetEntryPointArgCustom p mname)
     )
  => EntryPointRef mname
  -> IndigoM inp (ContractRef (GetEntryPointArgCustom p mname) & inp) ()
selfCalling epRef = nullaryOp (L.selfCalling @p epRef)

contractCalling
  :: forall inp cp epRef epArg addr exAddr.
     ( HasEntryPointArg cp epRef epArg
     , ToTAddress cp addr
     , ToT addr ~ ToT Address
     , IsExpr exAddr addr
     , TypeValue epArg
     )
  => epRef -> exAddr -> IndigoM inp (Maybe (ContractRef epArg) & inp) ()
contractCalling epRef addr = unaryOp addr (L.contractCalling @cp epRef)

-- | Give a name to given contract. Apply it to the whole contract code.
contractName :: Text -> IndigoM i o () -> IndigoM i o ()
contractName name b = IndigoM $ \md ->
  let GenCode _ mdb gc clr = runIndigoM b md in
  GenCode () mdb (L.contractName name gc) clr

-- Entrypoints

type IndigoContractCode param store =
     (HasStorage store, HasSideEffects)
  => Var param
  -> IndigoProcedure '[param, store, Ops]

type IndigoEntrypoint param = forall s. param -> IndigoProcedure s

transferTokens
  :: (IsExpr exp p, IsExpr exm Mutez, IsExpr exc (ContractRef p), NiceParameter p, HasSideEffects)
  => exp -> exm -> exc -> IndigoM inp inp ()
transferTokens ep em ec = do
  MetaData s _ <- iget
  ternaryOpFlat ep em ec (L.transferTokens # varActionOperation s)

setDelegate :: (HasSideEffects, IsExpr ex (Maybe KeyHash)) => ex -> IndigoM inp inp ()
setDelegate e =  do
  MetaData s _ <- iget
  unaryOpFlat e (L.setDelegate # varActionOperation s)

createContract
  ::
  ( TypeValue st
  , IsExpr exk (Maybe KeyHash), IsExpr exm Mutez, IsExpr exs st
  , NiceStorage st, NiceParameterFull param
  , HasSideEffects
  )
  => (HasStorage st => Var param -> IndigoM '[param, st, Ops] out ())
  -> exk
  -> exm
  -> exs
  -> IndigoM inp (Address & inp) (Var Address)
createContract iCtr ek em es =  do
  MetaData s _ <- iget
  ternaryOp ek em es (L.createContract (compileIndigoContract iCtr) # varActionOperation (NoRef :& s))
  makeTopVar

----------------------------------------------------------------------------
-- Functions and Procedures
----------------------------------------------------------------------------

-- | Utility type for an `IndigoM` that add one element to the stack and returns
-- a variable pointing at it.
type IndigoFunction s ret = IndigoM s (ScopeOutStack ret s) (ScopeRetVars ret)

-- | Utility type for an `IndigoM` that does not modify the stack (only the
-- values in it) and returns nothing.
type IndigoProcedure s = IndigoM s s ()

-- | Takes an arbitrary `IndigoM` and wraps it into an `IndigoFunction`
-- producing a local scope for its execution. Once it executed, all
-- non-returned variables are cleaned up so that the stack has only
-- returned variables at the top. This also can be interpreted as
-- `if True then f else nop`.
--
-- Note, that by default we do not define scope inside indigo functions,
-- meaning that once we want to create a new variable or return it from
-- a function we need to do it inside `scope $ instr` construction, for
-- example:
--
-- ```
-- f :: IndigoFunction s Natural
-- f = scope $ do
--   *[s]*
--   res <- newVar (0 :: Natural)
--   *[Natural, s]*
--   scope $ do
--     _n <- newVar (1 :: Integer)
--     *[Integer, Natural, s]
--     res += 4
--   *[Natural, s]*
--   return res
--   *[s]*
-- ```
scope
  :: forall a inp out . ScopeCodeGen a
  => IndigoM inp out a
  -> IndigoFunction inp a
scope f = IndigoM $ \md ->
  let gc = runIndigoM f md in
  finalizeStatement @a md (compileScope gc)

-- | Alias for `scope` we use in a tutorial.
defFunction
  :: forall a inp out . ScopeCodeGen a
  => IndigoM inp out a
  -> IndigoFunction inp a
defFunction = scope

-- | A more specific version of 'defFunction' meant to more easily create
-- 'IndigoContract's.
defContract
  :: ((HasStorage st, HasSideEffects) => IndigoM '[param, st, Ops] out ())
  -> ((HasStorage st, HasSideEffects) => IndigoProcedure '[param, st, Ops])
defContract = scope
