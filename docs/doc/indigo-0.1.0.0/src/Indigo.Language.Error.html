{-# OPTIONS_GHC -Wno-redundant-constraints #-}

-- | Error related statements of Indigo language.

module Indigo.Language.Error
  ( failWith
  , failUsing_
  , failCustom
  , failCustom_
  , failUnexpected_
  , assert
  ) where

import Indigo.Expr
import Indigo.Language.Conditional
import Indigo.Lorentz
import Indigo.Prelude
import Indigo.State (GenCode(..), IndigoM(..), iput, return, (>>))
import qualified Lorentz.Errors as L
import qualified Lorentz.Instr as L
import Util.Label (Label)

failIndigoM :: inp :-> out -> IndigoM inp out r
failIndigoM code = iput $ GenCode errOut errMd code failCl
  where
    -- note: here we can use errors for the output and MetaData, because they
    -- are lazy field of GenCode and, due to the way # combines the generated
    -- code (ignores everything following a failWith) they won't actually ever
    -- be accessed again. The same goes for the "cleaning" code, except it is
    -- not lazy and needs to typecheck, so we have to use `failWith` again.
    msg = " is undefined after a failing instruction"
    errOut = error $ "Output" <> msg
    errMd = error $ "MetaData" <> msg
    failCl = L.unit # L.failWith

failWith ::
  ( KnownValue a
  , IsExpr ex a
  )
  => ex -> IndigoM s t r
failWith exa = compileToExpr exa >> failIndigoM L.failWith

failUsing_ :: (IsError x) => x -> IndigoM s t r
failUsing_ x = failIndigoM (failUsing x)

failCustom
  :: forall tag err ex s t r.
     ( err ~ ErrorArg tag
     , CustomErrorHasDoc tag
     , NiceConstant err
     , IsExpr ex err
     )
  => Label tag -> ex -> IndigoM s t r
failCustom l errEx = withDict (niceConstantEvi @err) $ do
  compileToExpr errEx
  failIndigoM $ L.failCustom l

failCustom_
  :: forall tag s t r notVoidErrorMsg.
     ( RequireNoArgError tag notVoidErrorMsg
     , CustomErrorHasDoc tag
     )
  => Label tag -> IndigoM s t r
failCustom_ = failIndigoM . L.failCustom_

failUnexpected_ :: MText -> IndigoM s t r
failUnexpected_ msg = failUsing_ $ [mt|Unexpected: |] <> msg

assert
  :: forall s x ex.
  ( IsError x
  , IsExpr ex Bool
  )
  => x -> ex -> IndigoM s s ()
assert err e = if_ (toExpr e) (return ()) (failUsing_ err :: IndigoM s s ())
