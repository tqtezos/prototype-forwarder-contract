-- | This type is analogus to the `CaseClauseL` type but instead of wrapping a Lorentz
-- instruction, this wraps an Indigo value with the same input/output types.
data IndigoCaseClauseL (inp :: [Kind.Type]) ret (param :: CaseClauseParam) where
  IndigoCaseClauseL
    :: (AppendCtorField x inp ~ (ExtractCtorField x & inp))
    => (MetaData inp -> CaseClauseL inp (ScopeOutStack ret inp) ('CaseClauseParam ctor x))
    -> IndigoCaseClauseL inp ret ('CaseClauseParam ctor x)

data IndigoAnyOut x inp ret = forall xs retBranch .
  ( ScopeCodeGen retBranch
  , ScopeOutStack ret inp ~ ScopeOutStack retBranch inp
  ) =>
  IndigoAnyOut (IndigoM (x : inp) xs retBranch)

instance
  ( name ~ AppendSymbol "c" ctor
  , AppendCtorField field inp ~ (ExtractCtorField field & inp)
  , TypeValue (ExtractCtorField field)
  , x ~ ExtractCtorField field
  )
  =>
    CaseArrow
      name
      (Var x -> IndigoAnyOut x inp ret)
      (IndigoCaseClauseL inp ret ('CaseClauseParam ctor field))
  where
    (/->) _ ind =
      IndigoCaseClauseL (\(md :: MetaData inp) ->
        -- Create a reference to the top of stack
        let (varCase, mdCaseBody) = pushRefMd md in
        -- Pass the reference to the case body
        case ind varCase of
          IndigoAnyOut (body :: IndigoM (x : inp) _xs retBr) ->
            let gc = runIndigoM body mdCaseBody in
            CaseClauseL $
              -- Compute returning expressions and clean up everything
              compileScope gc #
              -- Remove @x@ from the stack too
              liftClear' @(ClassifyBranchReturn retBr) @retBr @(x & inp) @inp L.drop
      )

-- | An alias for /-> to use when the body is a lambda so that the
-- whole thing won't have multiple arrows.
(#=) :: CaseArrow name body clause => Label name -> body -> clause
(#=) a b = a /-> b

-- | Use this instead of /->.
-- This operator is like /-> but wrap a body into IndigoAnyOut,
-- which is needed for two reasons: to allow having any output stack
-- and to allow to return not exactly the same values.
(//->)
  :: ( CaseArrow name (Var x -> IndigoAnyOut x inp ret) (IndigoCaseClauseL inp ret ('CaseClauseParam ctor field))
     , ScopeCodeGen retBr
     , ret ~ ScopeRetExprs retBr
     , ScopeOutStack ret inp ~ ScopeOutStack retBr inp
     )
  => Label name
  -> (Var x -> IndigoM (x : inp) xs retBr)
  -> IndigoCaseClauseL inp ret ('CaseClauseParam ctor field)
(//->) a b = a /-> IndigoAnyOut <$> b
infixr 0 //->

-- This constraint is shared by all @case*@ functions.
type CaseCommon dt guard inp ret clauses =
  ( IsExpr guard dt
  , InstrCaseC dt inp inp
  , RMap (CaseClauses dt)
  , clauses ~ Rec (IndigoCaseClauseL inp ret) (CaseClauses dt)
  , ScopeCodeGen ret
  )

-- | A case statement for indigo. See examples for a sample usage.
case_
  :: forall dt guard inp ret clauses .
  (CaseCommon dt guard inp ret clauses)
  => guard
  -> clauses
  -> IndigoM inp (ScopeOutStack ret inp) (ScopeRetVars ret)
case_ g cls = IndigoM $ \md ->
  let cdG = gcCode $ runIndigoM (compileToExpr g) md in
  finalizeStatement @ret md (cdG # L.case_ (toCaseClauseL md cls))

-- | 'case_' for tuples.
caseT
  :: forall dt guard inp ret clauses.
  ( CaseCommon dt guard inp ret clauses
  , RecFromTuple clauses
  )
  => guard
  -> IsoRecTuple clauses
  -> IndigoM inp (ScopeOutStack ret inp) (ScopeRetVars ret)
caseT g = case_ g . recFromTuple @clauses

-- | 'case_' for pattern-matching on parameter.
entryCase_
  :: forall dt entryPointKind guard inp ret clauses .
  ( CaseCommon dt guard inp ret clauses
  , DocumentEntryPoints entryPointKind dt
  )
  => Proxy entryPointKind
  -> guard
  -> clauses
  -> IndigoM inp (ScopeOutStack ret inp) (ScopeRetVars ret)
entryCase_ proxy g cls = IndigoM $ \md ->
  let cdG = gcCode $ runIndigoM (compileToExpr g) md in
  finalizeStatement @ret md (cdG # L.entryCase_ proxy (toCaseClauseL md cls))

-- | 'entryCase_' for tuples.
entryCase
  :: forall dt entryPointKind guard inp ret clauses .
     ( CaseCommon dt guard inp ret clauses
     , RecFromTuple clauses
     , DocumentEntryPoints entryPointKind dt
     )
  => Proxy entryPointKind
  -> guard
  -> IsoRecTuple clauses
  -> IndigoM inp (ScopeOutStack ret inp) (ScopeRetVars ret)
entryCase proxy g = entryCase_ proxy g . recFromTuple @clauses

-- | 'entryCase' for contracts with flat parameter.
entryCaseSimple
  :: forall cp guard inp ret clauses .
     ( CaseCommon cp guard inp ret clauses
     , RecFromTuple clauses
     , DocumentEntryPoints PlainEntryPointsKind cp
     , NiceParameterFull cp
     , RequireFlatParamEps cp
     )
  => guard
  -> IsoRecTuple clauses
  -> IndigoM inp (ScopeOutStack ret inp) (ScopeRetVars ret)
entryCaseSimple g cls = IndigoM $ \md ->
  let cdG = gcCode $ runIndigoM (compileToExpr g) md in
  finalizeStatement @ret md (cdG # L.entryCaseSimple_ (toCaseClauseL md (recFromTuple @clauses cls)))

toCaseClauseL
  :: forall inp ret cs . (RMap cs)
  => MetaData inp
  -> Rec (IndigoCaseClauseL inp ret) cs
  -> Rec (CaseClauseL inp (ScopeOutStack ret inp)) cs
toCaseClauseL md r =
  rmap (\(IndigoCaseClauseL fn) -> fn md) r
