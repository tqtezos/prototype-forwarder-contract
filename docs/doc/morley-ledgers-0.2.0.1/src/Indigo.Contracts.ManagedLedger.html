-- | Transfers tokens between two given accounts.
transfer
  :: ( tp :~> AL.TransferParams
     , HasStorage Storage
     )
  => IndigoEntrypoint tp
transfer parameter = do
  doc $ DDescription L.transferDoc

  ensureNotPaused

  let
    to = parameter !. #to
    from = parameter !. #from
    value = parameter !. #value

  when (to /=. from) $ do

    when (sender /=. from) $ do
      consumeAllowance parameter

    let
      debitParam :: Expr BurnParams
      debitParam = constructT
        ( from !~ #from
        , value !~ #value
        )

      creditParam :: Expr MintParams
      creditParam = constructT
        ( to !~ #to
        , value !~ #value
        )

    debitFrom debitParam
    creditTo creditParam



-- | Allows spender to withdraw tokens from sender.
approve
  :: ( ap :~> ApproveParams
     , HasStorage Storage
     )
  => IndigoEntrypoint ap
approve parameter = scope $ do
  doc $ DDescription L.approveDoc

  ensureNotPaused

  let
    spender = parameter !. #spender
    value = parameter !. #value

  let
    allowanceParams :: Expr AllowanceParams
    allowanceParams = constructT
      ( sender !~ #owner
      , spender !~ #spender
      , value !~ #value
      )

  currentAllowance <- allowance allowanceParams

  when (currentAllowance /=. (0 :: Natural) &&. value /=. (0 :: Natural)) $
    failCustom #unsafeAllowanceChange currentAllowance

  setAllowance allowanceParams



-- | Compares the expected allowance value with the actual one
-- and sets a new one if they match.
approveCAS
  :: ( ap :~> ApproveCasParams
     , HasStorage Storage
     )
  => IndigoEntrypoint ap
approveCAS parameter = scope $ do
  doc $ DDescription L.approveCASDoc

  ensureNotPaused

  let
    spender = parameter !. #spender
    value = parameter !. #value
    expectedAllowance = parameter !. #expected

  let
    allowanceParams :: Expr AllowanceParams
    allowanceParams = constructT
      ( sender !~ #owner
      , spender !~ #spender
      , value !~ #value
      )

  actualAllowance <- allowance allowanceParams

  let
    allowanceMismatchError = pair
      (actualAllowance !~ #actual)
      (expectedAllowance !~ #expected)

  when (expectedAllowance /=. actualAllowance) $
    failCustom #allowanceMismatch allowanceMismatchError

  setAllowance allowanceParams



-- | Returns the approval value between two given addresses.
getAllowance
  :: ( ap :~> View AL.GetAllowanceParams Natural
     , HasStorage Storage
     , HasSideEffects
     )
  => IndigoEntrypoint ap
getAllowance parameter = scope $ do
  doc $ DDescription L.getAllowanceDoc

  let
    ledger_ = storage !. #ledger

  project parameter $ \param -> do

    let
      owner = param !. #owner
      spender = param !. #spender

    res <- newVar (0 :: Natural)

    whenJust (ledger_ &? owner) $ \value ->
      whenJust (value !. #approvals ^? spender) $
        setVar res

    return (varExpr res)



-- | Returns current balance of the addres in ledger.
getBalance
  :: ( ap :~> View AL.GetBalanceParams Natural
     , HasStorage Storage
     , HasSideEffects
     )
  => IndigoEntrypoint ap
getBalance parameter = scope $ do
  doc $ DDescription L.getBalanceDoc

  let
    ledger_ = storage !. #ledger

  project parameter $ \owner -> do
    res <- newVar (0 :: Natural)

    whenJust (ledger_ &? UnName #owner owner) $ \value ->
      setVar res (value !. #balance)

    return (varExpr res)


-- | Returns total number of tokens.
getTotalSupply
  :: ( tp :~> View () Natural
     , HasStorage Storage
     , HasSideEffects
     )
  => IndigoEntrypoint tp
getTotalSupply parameter = do
  doc $ DDescription L.getTotalSupplyDoc
  project parameter $ \_ ->
    return (storage !. #fields !. #totalSupply)



-- | Pauses transfer and approval operations if this
-- parameter is set to `False`.
setPause
  :: ( tp :~> Bool
     , HasStorage Storage
     )
  => IndigoEntrypoint tp
setPause parameter = scope $ do
  doc $ DDescription L.setPauseDoc
  fields <- newVar $ storage !. #fields
  setField_ fields #paused parameter
  setField_ storage #fields fields



-- | Changes current administrator.
setAdministrator
  :: ( sp :~> Address
     , HasStorage Storage
     )
  => IndigoEntrypoint sp
setAdministrator parameter = scope $ do
  doc $ DDescription L.setAdministratorDoc
  authorizeAdmin
  fields <- newVar $ storage !. #fields
  setField_ fields #admin parameter
  setField_ storage #fields fields



-- | Returns current administrator address.
getAdministrator
  :: ( gp :~> View () Address
     , HasStorage Storage
     , HasSideEffects
     )
  => IndigoEntrypoint gp
getAdministrator parameter = do
  doc $ DDescription L.getAdministratorDoc
  project parameter $ \_ ->
    return (storage !. #fields !. #admin)



-- | Produces token for the given address.
mint
  :: ( mp :~> MintParams
     , HasStorage Storage
     )
  => IndigoEntrypoint mp
mint parameter = do
  doc $ DDescription L.mintDoc
  authorizeAdmin
  creditTo parameter



-- | Reduces the given amount of tokens for the given address.
burn
  :: ( bp :~> BurnParams
     , HasStorage Storage
     )
  => IndigoEntrypoint bp
burn parameter = do
  doc $ DDescription L.burnDoc
  authorizeAdmin
  debitFrom parameter

-- | Debit the given amount of tokens from a given address in ledger.
debitFrom
  :: ( tp :~> BurnParams
     , HasStorage Storage
     )
  => IndigoEntrypoint tp
debitFrom parameter = scope $ do
  let
    from :: Expr Address
    from = parameter !. #from

  -- Balance check and the corresponding update.

  let
    ledger_ :: Expr (BigMap Address LedgerValue)
    ledger_ = storage !. #ledger

    val :: Expr Natural
    val = parameter !. #value

  let
    failNotEnoughBalance :: (ex :~> Natural) => ex -> IndigoProcedure s
    failNotEnoughBalance curBalance = failCustom #notEnoughBalance $ pair
      (val !~ #required)
      (curBalance !~ #present)

    updateLedger
      :: ( ev :~> LedgerValue
         , eb :~> Natural
         )
      => ev -> eb -> IndigoProcedure s
    updateLedger ledgerValue balance' = scope $ do
      newLedgerValue <- nonEmptyLedgerValue $ ledgerValue ~. (#balance, balance')
      whenJust (varExpr newLedgerValue) $ \ledgerValue' -> do
        setField_ storage #ledger $ ledger_ &~ (from, ledgerValue')

    whenExists :: Var LedgerValue -> IndigoProcedure s
    whenExists ledgerValue = scope $ do
      convertedBalance <- newVar (0 :: Natural)

      let oldBalance = ledgerValue !. #balance

      newBalance <- subGt0 oldBalance val (failNotEnoughBalance oldBalance)

      whenJust (varExpr newBalance) $
        setVar convertedBalance

      updateLedger ledgerValue convertedBalance
      addTotalSupply (neg val)

  ifJust (ledger_ &? from)
    whenExists
    (failNotEnoughBalance (0 :: Natural))



-- | Credit the given amount of tokens to a given address in ledger.
creditTo
  :: ( tp :~> MintParams
     , HasStorage Storage
     )
  => IndigoEntrypoint tp
creditTo parameter = do
  let
    to :: Expr Address
    to = parameter !. #to

    value :: Expr Natural
    value = parameter !. #value

    ledger_ :: Expr (BigMap Address LedgerValue)
    ledger_ = storage !. #ledger

  let
    updateStorage
      :: ( ev :~> LedgerValue
         , eb :~> Natural
         )
      => ev -> eb -> IndigoProcedure s
    updateStorage ledgerValue balance' = do
      let newLedgerValue = ledgerValue ~. (#balance, balance')
      setField_ storage #ledger $ ledger_ &~ (to, newLedgerValue)
      addTotalSupply (int value)

    whenExists
      :: ( ev :~> LedgerValue )
      => ev -> IndigoProcedure s
    whenExists ledgerValue = do
      let oldBalance = ledgerValue !. #balance
          newBalance = value +. oldBalance
      updateStorage ledgerValue newBalance
      -- TODO: somewhat if called here, it won't affect storage at all
      -- addTotalSupply storage (int value)

    whenDoesntExist =
      when (value /=. (0 :: Natural)) $ do
        updateStorage emptyLedgerValue value

  ifJust (ledger_ &? to)
    whenExists
    whenDoesntExist


-- | Return current allowance for spender.
allowance
  :: ( ap :~> AllowanceParams
     , HasStorage Storage
     )
  => ap -> IndigoFunction s Natural
  -- => ap -> IndigoM s (Natural ': s) (Var Natural)
allowance parameter = scope $ do

  let
    ledger_ = storage !. #ledger

  res <- newVar (0 :: Natural)

  let
    owner = parameter !. #owner
    spender = parameter !. #spender

  whenJust (ledger_ &? owner) $ \value ->
    whenJust (value !. #approvals ^? spender) $
      setVar res

  return res



-- | Set allowance for spender.
setAllowance
  :: ( ap :~> AllowanceParams
     , HasStorage Storage
     )
  => IndigoEntrypoint ap
setAllowance parameter = scope $ do
  let
    ledger_ = storage !. #ledger

  let
    owner   = parameter !. #owner
    spender = parameter !. #spender
    value   = parameter !. #value

  ledgerEntry <- newVar emptyLedgerValue

  whenJust (ledger_ &? owner) $
    setVar ledgerEntry

  let
    approvals = ledgerEntry !. #approvals
    updatedApprovals = approvals ^~ (spender, some value)
    newLedgerValue = ledgerEntry ~. (#approvals, updatedApprovals)

  setField_ storage #ledger $ ledger_ &~ (owner, newLedgerValue)



-- | Consume the given allowance from address.
consumeAllowance
  :: ( ap :~> TransferParams
     , HasStorage Storage
     )
  => IndigoEntrypoint ap
consumeAllowance parameter = scope $ do
  let
    from  = parameter !. #from
    value = parameter !. #value

  let
    allowanceParams :: Expr AllowanceParams
    allowanceParams = constructT
      ( from !~ #owner
      , sender !~ #spender
      , value !~ #value
      )

  currentAllowance <- allowance allowanceParams

  let
    updatedAllowance :: Expr (Maybe Natural)
    updatedAllowance = isNat $ currentAllowance -. value

  let
    setUpdatedAllowance updatedAmount = do
      let
        setAllowanceParams :: Expr AllowanceParams
        setAllowanceParams = constructT
          ( from !~ #owner
          , sender !~ #spender
          , updatedAmount !~ #value
          )

      setAllowance setAllowanceParams

    failNotEnoughAllowance :: IndigoProcedure s
    failNotEnoughAllowance = failCustom #notEnoughAllowance $ pair
      (value !~ #required)
      (currentAllowance !~ #present)

  ifJust updatedAllowance
    setUpdatedAllowance
    failNotEnoughAllowance



-- | Update total supply depending when integer amount
-- passed to it and fail if total supply is negative.
addTotalSupply
  :: ( amount :~> Integer
     , HasStorage Storage
     )
  => IndigoEntrypoint amount
addTotalSupply amountSupply = scope $ do
  fields <- newVar $ storage !. #fields

  let
    totalSupply = fields !. #totalSupply
    updatedSupply = isNat $ totalSupply +. amountSupply

  let
    setUpdatedSupply supply = do
      setField_ fields #totalSupply supply
      setField_ storage #fields fields

    failNegativeTotalSupply :: IndigoProcedure s
    failNegativeTotalSupply = failUnexpected_ [mt|Negative total supply|]

  ifJust updatedSupply
    setUpdatedSupply
    failNegativeTotalSupply



-- | Empty ledger value.
emptyLedgerValue :: Expr LedgerValue
emptyLedgerValue = constructT
  ( (constExpr 0) !~ #balance
  , (constExpr mempty) !~ #approvals
  )



-- | Ensure that given 'LedgerValue' value cannot be safely removed
-- and return it.
nonEmptyLedgerValue
  :: ( ledgerValue :~> LedgerValue )
  => ledgerValue -> IndigoFunction s (Maybe LedgerValue)
nonEmptyLedgerValue ledgerValue = scope $ do
  let
    curBalance = ledgerValue !. #balance
    approvals = ledgerValue !. #approvals

  res <- newVar none

  when (curBalance >=. (0 :: Natural)) $
    setVar res (some ledgerValue)

  when (msize approvals >=. (0 :: Natural)) $
    setVar res (some ledgerValue)

  return (varExpr res)



-- | Ensure that operations are not paused.
ensureNotPaused
  :: (HasStorage Storage)
  => IndigoProcedure s
ensureNotPaused = do
  let
    paused = storage !. #fields !. #paused

  when paused $
    failCustom_ #tokenOperationsArePaused



-- | Ensure that sender is admin.
authorizeAdmin
  :: (HasStorage Storage)
  => IndigoProcedure s
authorizeAdmin = do
  let
    currentAdmin = storage !. #fields !. #admin
  when (sender /=. currentAdmin) $
    failCustom_ #senderIsNotAdmin


----------------------------------------------------------------------------
-- Contract
----------------------------------------------------------------------------

managedLedgerContract :: ContractCode Parameter Storage
managedLedgerContract = compileIndigoContract managedLedgerIndigo

storage :: HasStorage Storage => Var Storage
storage = storageVar

managedLedgerIndigo :: IndigoContractCode Parameter Storage
managedLedgerIndigo param = contractName "Managed Ledger" $ do
  doc $ $mkDGitRevision morleyRepoSettings
  doc $ DDescription contractDoc
  entryCaseSimple param
    ( #cTransfer //-> transfer
    , #cApprove //-> approve
    , #cApproveCAS //-> approveCAS
    , #cGetAllowance //-> getAllowance
    , #cGetBalance //-> getBalance
    , #cGetTotalSupply //-> getTotalSupply
    , #cSetPause //-> setPause
    , #cSetAdministrator //-> setAdministrator
    , #cGetAdministrator //-> getAdministrator
    , #cMint //-> mint
    , #cBurn //-> burn
    )
  where
    contractDoc :: Markdown
    contractDoc =
      "This documentatiwhen describes a managed ledger contract " <>
      "that follows TZIP-FA1."
