module Morley.Client.Parser
  ( ClientArgs (..)
  , ClientArgsRaw (..)
  , OriginateArgs (..)
  , TransferArgs (..)
  , NodeInfoRef (..)
  , argParser
  , nodeInfoOptions
  , nodeInfoRefOptions

    -- * Misc
  , getFromNodeInfoRef
  ) where

import qualified Data.Aeson as Aeson
import Fmt (pretty)
import Options.Applicative
  (auto, eitherReader, help, long, metavar, option, short, str, strOption, subparser, switch)
import qualified Options.Applicative as Opt

import qualified Michelson.Parser as P
import qualified Michelson.Untyped as U
import Morley.Client.Types
import Tezos.Address
import qualified Tezos.Crypto.Ed25519 as Ed25519 (SecretKey, parseSecretKey)
import Util.Exception

data ClientArgs
  = ClientArgs NodeInfo ClientArgsRaw

data ClientArgsRaw
  = Originate SenderInfo OriginateArgs
  | Transfer SenderInfo TransferArgs

data OriginateArgs = OriginateArgs
  { oaMbContractFile :: !(Maybe FilePath)
  , oaInitialBalance :: !Natural
  , oaInitialStorage :: !U.Value
  }

data TransferArgs = TransferArgs
  { taDestination :: !Address
  , taAmount :: !Natural
  , taParameter :: !U.Value
  }

data NodeInfoRef
  = NodeInfoGiven NodeInfo
  | NodeInfoFromConfig FilePath

getFromNodeInfoRef :: (MonadIO m, MonadThrow m) => NodeInfoRef -> m NodeInfo
getFromNodeInfoRef = \case
  NodeInfoGiven ni -> pure ni
  NodeInfoFromConfig path -> do
    res <- liftIO $ Aeson.eitherDecodeFileStrict path
    either (throwM . TextException . toText) pure res

argParser :: Opt.Parser ClientArgs
argParser = ClientArgs <$> nodeInfoOptions <*> argsRawParser

nodeInfoOptions :: Opt.Parser NodeInfo
nodeInfoOptions = NodeInfo <$>
  urlOption <*> portOption <*> useHttpsSwitch
  where
    useHttpsSwitch :: Opt.Parser Bool
    useHttpsSwitch = switch $
       short 'S' <> long "use-https" <>
       help "Use HTTPS to communicate with the remote node"

nodeInfoRefOptions :: Opt.Parser NodeInfoRef
nodeInfoRefOptions = foldr1 (<|>)
  [ NodeInfoGiven <$> nodeInfoOptions
  , NodeInfoFromConfig <$> configPathOption
  ]
  where
    configPathOption = strOption $
      long "node-info-config" <>
      metavar "FILEPATH" <>
      help "Path to tezos-client config."

urlOption :: Opt.Parser Text
urlOption = option str $
  mconcat [ short 'A', long "node-url", metavar "URL"
          , help "Remote node URL"
          ]

portOption :: Opt.Parser Int
portOption = option auto $
  mconcat [ short 'P', long "node-port", metavar "PORT"
          , help "Remote node port"
          ]

senderInfoOptions :: Opt.Parser SenderInfo
senderInfoOptions = SenderInfo <$>
  namedAddressOption "from"
  "The address from which operation is performed" <*>
  namedSecretKeyOption "secret-key"
  "The secret key of the operation performer"

namedAddressOption :: String -> String -> Opt.Parser Address
namedAddressOption longName hInfo = option (eitherReader parseAddrDo) $
  mconcat
    [ metavar "ADDRESS"
    , long longName
    , help hInfo
    ]
  where
    parseAddrDo :: String -> Either String Address
    parseAddrDo addr =
      either (Left . mappend "Failed to parse address: " . pretty) Right $
      parseAddress $ toText addr

namedSecretKeyOption :: String -> String -> Opt.Parser Ed25519.SecretKey
namedSecretKeyOption longName hInfo = option (eitherReader parseSecretKeyDo) $
  mconcat
    [ metavar "SECRET_KEY"
    , long longName
    , help hInfo
    ]
  where
    parseSecretKeyDo :: String -> Either String Ed25519.SecretKey
    parseSecretKeyDo addr =
      either (Left . mappend "Failed to parse secret key: " . pretty) Right $
      Ed25519.parseSecretKey
      $ toText addr

argsRawParser :: Opt.Parser ClientArgsRaw
argsRawParser = subparser $
  originateCmd <> transferCmd
  where
    mkCommandParser commandName parser desc =
      Opt.command commandName $
      Opt.info (Opt.helper <*> parser) $
      Opt.progDesc desc
    originateCmd =
      mkCommandParser "originate"
      (Originate <$> senderInfoOptions <*> originateArgsOption)
      "Originate passed contract on real network"
    transferCmd =
      mkCommandParser "transfer"
      (Transfer <$> senderInfoOptions <*> transferArgsOption)
      "Perform a transfer to the given contract with given amount and parameter"
    originateArgsOption :: Opt.Parser OriginateArgs
    originateArgsOption = OriginateArgs <$>
      ( optional $ strOption $
        long "contract" <>
        metavar "FILEPATH" <>
        help "Path to contract file"
      ) <*>
      naturalOption 0 "initial-balance" "Inital balance of the contract" <*>
      valueOption "initial-storage" "Initial contract storage value"
    transferArgsOption :: Opt.Parser TransferArgs
    transferArgsOption = TransferArgs <$>
      namedAddressOption "to" "Address of the contract that receives transfer" <*>
      naturalOption 0 "amount" "Transfer amount" <*>
      valueOption "parameter" "Transfer parameter"

valueOption :: String -> String -> Opt.Parser U.Value
valueOption name hInfo = option (eitherReader parseValue) $
  long name <>
  help hInfo
  where
    parseValue :: String -> Either String U.Value
    parseValue =
      first (mappend "Failed to parse value: " . displayException) .
      P.parseExpandValue .
      toText

naturalOption :: Natural -> String -> String -> Opt.Parser Natural
naturalOption defaultValue name hInfo =
  option auto $
  long name <>
  metavar "NAT" <>
  Opt.value defaultValue <>
  help hInfo
