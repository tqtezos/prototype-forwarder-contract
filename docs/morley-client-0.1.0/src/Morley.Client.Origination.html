-- | Originate given contract with given initial storage. Returns
-- operation hash, block in which this operation appears and originated contract
-- address.
originateContract
  :: forall m cp st. HasTezosNodeRpc m
  => SenderInfo -> FullContract cp st -> Value st
  -> m (Text, Text, Address)
originateContract SenderInfo{..} contract initialStorage = do
  OperationConstants{..} <- preProcessOperation siAddress
  let origOp = OriginationOperation
        { ooKind = "origination"
        , ooSource = siAddress
        , ooFee = 50000
        , ooCounter = ocCounter + 1
        , ooGasLimit = 800000
        , ooStorageLimit = 60000
        , ooBalance = 0
        , ooScript = mkOriginationScript contract initialStorage
        }
  let runOp = RunOperation
        { roOperation =
          RunOperationInternal
          { roiBranch = ocLastBlockHash
          , roiContents = [Right origOp]
          , roiSignature = stubSignature
          }
        , roChainId = bcChainId ocBlockConstants
        }
  -- Estimate limits and paid storage diff
  -- We run only one op and expect to have only one result
  ars1 <- getAppliedResults (Left runOp)
  ar1 <- handleAppliedResults ars1

  -- Update limits and fee
  let updOrigOp = origOp { ooGasLimit = arConsumedGas ar1 + 200
                         , ooStorageLimit = arStorageSize ar1 + 1000
                         , ooFee = calcFees (arConsumedGas ar1) (arPaidStorageDiff ar1)
                         }
  HexJSONByteString hex <- forgeOperation $ ForgeOperation
    { foBranch = ocLastBlockHash
    , foContents = [Right updOrigOp]
    }
  let signature' = Ed25519.sign siSecretKey (addOperationPrefix hex)
  let preApplyOp = PreApplyOperation
        { paoProtocol = bcProtocol ocBlockConstants
        , paoBranch = ocLastBlockHash
        , paoContents = [Right updOrigOp]
        , paoSignature = signature'
        }
  -- Perform operations preapplying in order to obtain contract address
  -- We preapply only one op and expect to have only one result
  ars2 <- getAppliedResults (Right preApplyOp)
  ar2 <- handleAppliedResults ars2
  contractAddr <- case arOriginatedContracts ar2 of
    [contractAddr'] -> pure contractAddr'
    contracts -> throwM $ PreApplyOriginatedMoreContracts contracts
  operationHash <- injectOperation $ prepareOpForInjection hex signature'
  blockHash <- waitForOperationInclusion operationHash
  pure (operationHash, blockHash, contractAddr)
    where
      handleAppliedResults :: [AppliedResult] -> m AppliedResult
      handleAppliedResults appliedResults = case appliedResults of
            [ar] -> pure ar
            _ -> throwM $ PreApplyUnexpectedSizeResult 1 $ length appliedResults

-- | Lorentz version of 'originateContract'
lOriginateContract
  :: (HasTezosNodeRpc m, NiceStorage st, NiceParameterFull cp)
  => SenderInfo -> L.ContractCode cp st -> st
  -> m (Text, Text, Address)
lOriginateContract si lContract lStorage =
  originateContract si (L.compileLorentzContract lContract) (L.toVal lStorage)

mkOriginationScript
  :: FullContract cp st -> Value st -> OriginationScript
mkOriginationScript FullContract{..} initialStorage = OriginationScript
  { osCode = Expression_Seq $ fromList
    [ Expression_Prim $
      MichelinePrimAp (MichelinePrimitive "parameter")
      (fromList [toExpression $ unParamNotes $ fcParamNotesSafe])
      (fromList [])
    , Expression_Prim $
      MichelinePrimAp (MichelinePrimitive "storage")
      (fromList [toExpression $ fcStoreNotes])
      (fromList [])
    , Expression_Prim $
      MichelinePrimAp (MichelinePrimitive "code")
      (fromList [toExpression fcCode])
      (fromList [])
    ]
  , osStorage = toExpression initialStorage
  }
