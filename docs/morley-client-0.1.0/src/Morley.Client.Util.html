module Morley.Client.Util
  ( calcFees
  , prepareOpForInjection
  , throwLeft
  , decodeOriginationScriptCode
  ) where

import qualified Data.Binary.Get as Get
import qualified Data.ByteString as BS (ByteString)
import qualified Data.ByteString.Lazy as LBS
import Text.Hex (encodeHex)
import Tezos.Common.Binary (encode)
import Tezos.Common.Json (TezosInt64)

import Michelson.Interpret.Unpack
import qualified Michelson.Untyped as U
import Morley.Client.Types
import qualified Tezos.Crypto.Ed25519 as Ed25519 (Signature, signatureToBytes)

calcFees :: TezosInt64 -> TezosInt64 -> TezosInt64
calcFees consumedGas storageSize =
  minimalFees +
  minimalTezPerGas * consumedGas +
  minimalTezPerStorage * storageSize
  where
    minimalFees = 100
    minimalTezPerGas = 1
    minimalTezPerStorage = 10

prepareOpForInjection :: BS.ByteString -> Ed25519.Signature -> Text
prepareOpForInjection operationHex signature' =
  (encodeHex operationHex) <> (encodeHex . Ed25519.signatureToBytes) signature'

throwLeft :: (MonadThrow m, Exception e) => m (Either e a) -> m a
throwLeft =
  (>>= \case
      Left e -> throwM e
      Right x -> return x)


decodeOriginationScriptCode :: OriginationScript -> Either UnpackError U.Contract
decodeOriginationScriptCode OriginationScript{..} =
  case Get.runGetOrFail decodeContract (LBS.fromStrict $ encode osCode) of
    Left (_, _, err) -> Left $ UnpackError $ toText err
    Right (_, _, res) -> Right res
